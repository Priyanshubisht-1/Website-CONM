#include <stdio.h>
#include <math.h>

float binaryToFloat(const char* binary) {
    float result = 0.0;
    int integerPart = 1;
    float fractionalValue = 0.5;

    while (*binary) {
        if (*binary == '.') {
            integerPart = 0;
        } else {
            if (integerPart) {
                result = result * 2 + (*binary - '0');
            } else {
                result += (*binary - '0') * fractionalValue;
                fractionalValue /= 2;
            }
        }
        binary++;
    }

    return result;
}

int main() {
    char binary[50];

    printf("Enter the binary number : ");
    scanf("%s", binary);

    float result = binaryToFloat(binary);
    printf("The float representation is: %f\n", result);

    return 0;
}

#include <stdio.h>
#include <math.h>
#include <conio.h>

int expo[10], coeff[10], n;
int dexpo[10], dcoeff[10], dn;

float fx(float x)
{
    float sum = 0;
    for (int i = 0; i < n; i++)
    {
        sum += coeff[i] * pow(x, expo[i]);
    }
    return sum;
}

float fdx(float x)
{
    float sum = 0;
    for (int i = 0; i < dn; i++)
    {
        sum += dcoeff[i] * pow(x, dexpo[i]);
    }
    return sum;
}

void NewtonRaphson()
{
    int i = 0, fla = 0, flb = 0;
    float a, b, x, e, fxa, fxb;
    for (int i = -3; i < 5; i++)
    {
        e = fx(i);
        if (e < 0)
        {
            a = i;
            fxa = e;
            fla = 1;
        }
        else if (e > 0)
        {
            b = i;
            fxb = e;
            flb = 1;
        }

        if (fla == 1 && flb == 1)
            break;
    }

    printf("\na : %f and b : %f\n", a, b);
    printf("\nf(a) : %f and f(b) : %f\n", fxa, fxb);

    x = (a + b) / 2;
    // x = 2.625;
    printf("Value of x : %f", x);
    float m, me;
    do
    {
        m = x - ((fx(x)) / (fdx(x)));
        me = fx(m);
        printf("\nIteration %d", ++i);
        printf("\nx : %f and f(x) : %f\n", m, me);
        x = m;
    } while (me > 0.0002 || me < -0.0002);

}

    int k = 2;
    do
    {
        pp = cp;
        for (int i = 1; i <= expo[0]; i++)
            b[i] = a[i] + b[i - 1] * pp;

        printf("\n");
        for (int i = 0; i <= expo[0]; i++)
            printf("b%d : %f -- ", i, b[i]);

        printf("\n");
        for (int i = 1; i < expo[0]; i++)
            c[i] = b[i] + c[i - 1] * pp;

        for (int i = 0; i < expo[0]; i++)
            printf("c%d : %f -- ", i, c[i]);

        cp = pp - b[expo[0]] / c[expo[0] - 1];
        printf("\n %d Iteration p%d : %f", k, k++, cp);
        printf("\n");
    } while (cp - pp < -0.002 || cp - pp > 0.002);

    return 0;
}

Gauss Elimination Method

Code :

#include <stdio.h>
#include <conio.h>

int a[3][3], c[3];

void GaussElimination()
{
    float x, y, z;
    for (int i = 0; i < 2; i++)
    {
        for (int j = i + 1; j < 3; j++)
        {
            int fact1 = a[i][i], fact2 = a[j][i];
            for (int k = 0; k < 3; k++)
                a[j][k] = fact1 * a[j][k] - fact2 * a[i][k];
            c[j] = fact1 * c[j] - fact2 * c[i];
        }
    }
    z = (float)c[2] / a[2][2];
    y = (float)(c[1] - (a[1][2] * z)) / a[1][1];
    x = (float)(c[0] - (a[0][1] * y) - (a[0][2] * z)) / a[0][0];

    printf("\nSolution\nx : %f\ny : %f\nz : %f\n", x, y, z);
}

void display()
{
    for (int i = 0; i < 3; i++)
        printf("\nEquation %d : %dx + %dy + %dz = %d", i + 1,
                   a[i][0], a[i][1], a[i][2], c[i]);
    printf("\n");
}

int main()
{
    for (int i = 0; i < 3; i++)
    {
        printf("Enter the %d Equation 's Coefficient  : ", i + 1);
        scanf("%d %d %d %d", &a[i][0], &a[i][1], &a[i][2], &c[i]);
    }
    display();
    GaussElimination();
    // display();

    return 0;
}

Gauss Jordan Method

#include <stdio.h>
#include <conio.h>

int a[3][3], c[3];

void GaussJordan()
{
    float x, y, z;
    for (int i = 0; i < 2; i++)
    {
        for (int j = i + 1; j < 3; j++)
        {
            int fact1 = a[i][i], fact2 = a[j][i];
            for (int k = 0; k < 3; k++)
                a[j][k] = fact1 * a[j][k] - fact2 * a[i][k];
            c[j] = fact1 * c[j] - fact2 * c[i];
        }
    }

    for (int i = 2; i > 0; i--)
    {
        for (int j = i - 1; j >= 0; j--)
        {
            int fact1 = a[i][i], fact2 = a[j][i];
            for (int k = 0; k < 3; k++)
                a[j][k] = fact1 * a[j][k] - fact2 * a[i][k];
            c[j] = fact1 * c[j] - fact2 * c[i];
        }
    }

    z = (float)c[2] / a[2][2];
    y = (float)c[1] / a[1][1];
    x = (float)c[0] / a[0][0];

    printf("\nSolution\nx : %f\ny : %f\nz : %f\n", x, y, z);
}

void display()
{
    for (int i = 0; i < 3; i++)
        printf("\nEquation %d : %dx + %dy + %dz = %d", i + 1, 
                a[i][0], a[i][1], a[i][2], c[i]);
    printf("\n");
}

int main()
{
    for (int i = 0; i < 3; i++)
    {
        printf("Enter the %d Equation 's Coefficient  : ", i + 1);
        scanf("%d %d %d %d", &a[i][0], &a[i][1], &a[i][2], &c[i]);
    }
    display();
    GaussJordan();
    // display();

    return 0;
}

#include <stdio.h>

void display()
{
    for (int i = 0; i < 3; i++)
        printf("Equation %d : %dx + %dy + %dz = %d\n", i + 1,

}

void GaussJacobi()
{
    int i = 1;
    float x, y, z, px, py, pz;
    x = y = z = 0;
    do
    {
        px = x;
        py = y;
        pz = z;
        x = (float)(c[0] - (a[0][1] * py) - (a[0][2] * pz)) / a[0][0];
        y = (float)(c[1] - (a[1][0] * px) - (a[1][2] * pz)) / a[1][1];
        z = (float)(c[2] - (a[2][0] * px) - (a[2][1] * py)) / a[2][2];
        printf("\nIteration %d : \nx : %f\ny : %f\nz : %f\n", i++, x, y, z);
    } while (px != x && py != y && pz != z);
    printf("\nSolution\nx : %f\ny : %f\nz : %f\n", x, y, z);
}

{
    for (int i = 0; i < 3; i++)

Lagrange Interpolation

Code :

#include <stdio.h>
#include <conio.h>

int n=3;
float x[10],fx[10];

float Interpolation (float gx)
{
    float gfx=0.0;
    for(int i=0; i < n; i++)
    {
        int j;
        float tynum=1.0, tyden=1.0;
        for(j=0; j < n; j++)
        {
            if(i!=j)
            {
                tynum*= (gx - x[j]);
                tyden*= (x[i]-x[j]);
                //printf("\npgx : %f and xj : %f",gx,x[j]);
                //printf("\nxi : %f and xj : %f",x[i],x[j]);
            }
        }
        //printf("\ntynum : %f , tyden : %f",tynum,tyden);
        gfx+=((tynum/tyden)*fx[i]);
    }
    return gfx;
}

int main()
{
    printf ("Enter the number of terms : ");
    scanf("%d",&n);
    for(int i=0; i < n; i++)
    {
        printf("Enter the %d term: ",i+1);
        scanf("%f %f",&x[i],&fx[i]);
    }

    float gx=0.0;
    printf("Enter the value of x : ");
    scanf("%f",&gx);

    printf("Tx at %f is %f\n", gx, Interpolation(gx));
    return 0;
}

#include <conio.h>

int n=3;
float x[10],fx[10];

float InverseInterpolation (float yx)
{
    float gx=0.0;
    for(int i=0; i < ;n; i++)
    {
        int j;
        float tynum=1.0, tyden=1.0;
        for(j=0; j < n; j++)
        {
            if(i!=j)
            {
                tynum*= (yx - fx[j]);
                tyden*= (fx[i]-fx[j]);
                //printf("\npgx : %f and xj : %f",gx,x[j]);
